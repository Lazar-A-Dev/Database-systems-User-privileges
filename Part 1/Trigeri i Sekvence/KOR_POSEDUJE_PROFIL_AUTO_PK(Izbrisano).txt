CREATE SEQUENCE  "KOR_POSEDUJE_PROFIL_KOD_SEQ"  MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 11 CACHE 20 ORDER  NOCYCLE ;

CREATE OR REPLACE TRIGGER "KOR_POSEDUJE_PROFIL_AUTO_PK" 
    BEFORE INSERT
    ON KOR_POSEDUJE_PROFIL
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN
    SELECT KOR_POSEDUJE_PROFIL_KOD_SEQ.NEXTVAL INTO :NEW.KOD FROM DUAL;
END;
