CREATE SEQUENCE  "SPISAK_IP_KOD_SEQ"  MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 3 START WITH 16107 CACHE 20 ORDER  NOCYCLE ;

CREATE OR REPLACE TRIGGER "SPISAK_IP_AUTO_PK" 
    BEFORE INSERT
    ON SPISAK_IP_ADRESA
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN
    SELECT SPISAK_IP_KOD_SEQ.NEXTVAL INTO :NEW.KOD FROM DUAL;
END;
