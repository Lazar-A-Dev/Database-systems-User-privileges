CREATE SEQUENCE  "PROFIL_REDNIBR_SEQ"  MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 17244 CACHE 20 ORDER  NOCYCLE ;

CREATE OR REPLACE TRIGGER "PROFIL_AUTO_PK" 
    BEFORE INSERT
    ON PROFIL
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN
    SELECT PROFIL_REDNIBR_SEQ.NEXTVAL INTO :NEW.REDNIBR FROM DUAL;
END;
