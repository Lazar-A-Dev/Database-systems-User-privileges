CREATE OR REPLACE TRIGGER "SISTEM_PROVERE_AUTO_PK" 
    BEFORE INSERT
    ON SISTEM_PROVERE
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN
    SELECT SISTEM_PROVERE_KOD_SEQ.NEXTVAL INTO :NEW.KOD FROM DUAL;
END;
