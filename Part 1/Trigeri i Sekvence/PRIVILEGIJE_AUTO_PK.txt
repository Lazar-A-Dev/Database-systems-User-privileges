CREATE SEQUENCE  "PRIVILEGIJE_BROJ_SEQ"  MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 5 CACHE 20 ORDER  NOCYCLE ;

CREATE OR REPLACE TRIGGER "PRIVILEGIJE_AUTO_PK" 
    BEFORE INSERT
    ON PRIVILEGIJE
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN
    SELECT PRIVILEGIJE_BROJ_SEQ.NEXTVAL INTO :NEW.BROJ FROM DUAL;
END;
