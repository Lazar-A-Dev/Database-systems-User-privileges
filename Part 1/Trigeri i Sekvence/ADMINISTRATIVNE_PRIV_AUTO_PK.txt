CREATE SEQUENCE  "ADMIN_PRIV_BROJ_SEQ"  MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 4 CACHE 20 ORDER  NOCYCLE ;

CREATE OR REPLACE TRIGGER "ADMIN_PRIV_AUTO_PK" 
    BEFORE INSERT
    ON ADMINISTRATIVNE_PRIV
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN
    SELECT ADMIN_PRIV_BROJ_SEQ.NEXTVAL INTO :NEW.BROJ FROM DUAL;
END;
