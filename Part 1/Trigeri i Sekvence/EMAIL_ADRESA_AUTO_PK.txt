CREATE SEQUENCE  "EMAIL_ADRESA_KOD_SEQ"  MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 16091 CACHE 20 ORDER  NOCYCLE ;

CREATE OR REPLACE TRIGGER "EMAIL_ADRESA_AUTO_PK" 
    BEFORE INSERT
    ON EMAIL_ADRESA
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN
    SELECT EMAIL_ADRESA_KOD_SEQ.NEXTVAL INTO :NEW.KOD FROM DUAL;
END;
