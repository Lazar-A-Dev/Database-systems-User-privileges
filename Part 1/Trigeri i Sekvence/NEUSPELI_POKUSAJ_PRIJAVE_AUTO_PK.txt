CREATE SEQUENCE  "NPP_KOD_SEQ"  MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 14521 CACHE 20 ORDER  NOCYCLE ;

CREATE OR REPLACE TRIGGER "NPP_AUTO_PK" 
    BEFORE INSERT
    ON NEUSPELI_POKUSAJ_PRIJAVE
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN
    SELECT NPP_KOD_SEQ.NEXTVAL INTO :NEW.KOD FROM DUAL;
END;
