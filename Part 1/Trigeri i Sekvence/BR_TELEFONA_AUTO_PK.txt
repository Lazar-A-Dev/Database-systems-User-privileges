CREATE SEQUENCE  "BR_TELEFONA_KOD_SEQ"  MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 16107 CACHE 20 ORDER  NOCYCLE ;

CREATE OR REPLACE TRIGGER "BR_TELEFONA_AUTO_PK" 
    BEFORE INSERT
    ON BR_TELEFONA
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN
    SELECT BR_TELEFONA_KOD_SEQ.NEXTVAL INTO :NEW.KOD FROM DUAL;
END;
