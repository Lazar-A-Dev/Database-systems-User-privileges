CREATE SEQUENCE  "ROD_STAV_MENIJA_KOD_SEQ"  MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 2 CACHE 20 ORDER  NOCYCLE ;

CREATE OR REPLACE TRIGGER "ROD_STAV_MENIJA_AUTO_PK" 
    BEFORE INSERT
    ON ROD_STAVKA_MENIJA
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN
    SELECT ROD_STAV_MENIJA_KOD_SEQ.NEXTVAL INTO :NEW.KOD FROM DUAL;
END;
