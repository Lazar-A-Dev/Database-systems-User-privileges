CREATE SEQUENCE  "SPISAK_IP_GRUPE_KOD_SEQ"  MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 14599 CACHE 20 ORDER  NOCYCLE ;

CREATE OR REPLACE TRIGGER "SPISAK_IP_GRUPE_AUTO_PK" 
    BEFORE INSERT
    ON SPISAK_IP_ADRESA_GRUPE
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN

END;
